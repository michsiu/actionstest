<script>
  let ws = null;
  let currentBotMsg = null;
  const messages = document.getElementById('messages');
  const input = document.getElementById('input');
  const fileInput = document.getElementById('fileInput');
  const wssInput = document.getElementById('wssUrl');
  const statusBar = document.getElementById('statusBar');
  const connectDiv = document.getElementById('connect');

  function scroll() { 
    requestAnimationFrame(() => {
      messages.scrollTop = messages.scrollHeight;
    });
  }

  function addMsg(type, text = '') {
    const d = document.createElement('div');
    d.className = 'msg ' + (type === 'user' ? 'user' : 'bot');
    if (text) d.textContent = text;
    messages.appendChild(d);
    scroll();
    return d;
  }

  function updateStatus(connected) {
    if (connected) {
      statusBar.classList.remove('disconnected');
    } else {
      statusBar.classList.add('disconnected');
    }
  }

  function handleServerMsg(data) {
    try {
      const { text, stream } = data;
      if (stream) {
        if (!currentBotMsg) {
          currentBotMsg = addMsg('bot');
        }
        currentBotMsg.textContent = text;
      } else {
        if (currentBotMsg) {
          currentBotMsg.textContent = text;
          currentBotMsg = null;
        } else {
          addMsg('bot', text);
        }
      }
      scroll();
    } catch(e) {
      addMsg('bot', '⚠️ 消息解析错误: ' + e.message);
    }
  }

  function connectWSS() {
    const url = wssInput.value.trim();
    if (!url) return;

    // 关闭已有连接
    if (ws) {
      ws.close();
      ws = null;
    }

    ws = new WebSocket(url);

    ws.onopen = () => {
      updateStatus(true);
      addMsg('bot', '✅ 已连接到转录服务');
      input.disabled = false;
      connectDiv.classList.add('hidden');
    };

    ws.onmessage = (e) => {
      try {
        const data = JSON.parse(e.data);
        handleServerMsg(data);
      } catch {
        addMsg('bot', e.data);
      }
    };

    ws.onerror = () => {
      addMsg('bot', '❌ WebSocket 连接错误');
    };

    ws.onclose = () => {
      updateStatus(false);
      addMsg('bot', '🔌 连接已断开');
      input.disabled = true;
      connectDiv.classList.remove('hidden');
      currentBotMsg = null;
      ws = null;
    };
  }

  // 连接按钮
  document.getElementById('connectBtn').onclick = connectWSS;
  wssInput.addEventListener('keydown', e => {
    if (e.key === 'Enter') connectWSS();
  });

  // 发送消息
  document.getElementById('send').onclick = () => {
    if (!ws || ws.readyState !== 1) {
      addMsg('bot', '⚠️ 未连接到服务');
      return;
    }
    const text = input.value.trim();
    if (!text) return;

    currentBotMsg = null;
    addMsg('user', text);
    ws.send(text);
    input.value = '';
  };

  // 回车发送
  input.addEventListener('keydown', e => {
    if (e.key === 'Enter') document.getElementById('send').click();
  });

  // 结束按钮（修正版）
  document.getElementById('end').onclick = () => {
    if (ws && ws.readyState === 1) {
      currentBotMsg = null;
      ws.send('/bye');
      addMsg('bot', '👋 会话已结束');
      input.disabled = true;
      // 延迟关闭，确保服务端收到 /bye
      setTimeout(() => {
        if (ws) {
          ws.close();
          ws = null;
          updateStatus(false);
          connectDiv.classList.remove('hidden');
        }
      }, 1000);
    }
  };

  // 文件上传按钮
  document.getElementById('addFileBtn').onclick = () => fileInput.click();

  // 文件选择处理
  fileInput.addEventListener('change', async (e) => {
    const file = e.target.files[0];
    if (!file || !ws || ws.readyState !== 1) {
      if (!ws || ws.readyState !== 1) {
        addMsg('bot', '⚠️ 未连接到服务');
      }
      fileInput.value = '';
      return;
    }

    currentBotMsg = null;

    const sizeMB = (file.size / 1024 / 1024).toFixed(2);
    addMsg('user', `📎 ${file.name}\n(${sizeMB} MB)`);

    try {
      ws.send(JSON.stringify({ type: 'file', filename: file.name }));
      const buffer = await file.arrayBuffer();
      ws.send(buffer);
      addMsg('bot', '📤 文件已发送，等待处理...');
    } catch (e) {
      addMsg('bot', '❌ 文件发送失败: ' + e.message);
    }

    fileInput.value = '';
  });

  // 页面关闭时发送 /bye
  window.addEventListener('beforeunload', () => {
    if (ws && ws.readyState === 1) {
      ws.send('/bye');
    }
  });

  // 初始状态
  updateStatus(false);
</script>
